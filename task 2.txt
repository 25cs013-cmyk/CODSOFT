    <script>
        // Store for local quizzes
        let quizzes = [
            { id: 1, title: "Web Basics", questions: [{text: "What does HTML stand for?", a: "HyperText Markup Language", b: "Home Tool Markup Language", correct: "A"}]}
        ];

        function showView(viewId) {
            document.getElementById('home-view').classList.add('hidden');
            document.getElementById('create-view').classList.add('hidden');
            document.getElementById('take-view').classList.add('hidden');
            document.getElementById(viewId).classList.remove('hidden');
            if(viewId === 'home-view') renderQuizList();
        }

        function renderQuizList() {
            const container = document.getElementById('quiz-list-container');
            container.innerHTML = quizzes.map(q => `
                <div class="card" style="display: flex; justify-content: space-between; align-items: center;">
                    <span><strong>${q.title}</strong></span>
                    <button class="btn btn-create" onclick="startQuiz(${q.id})">Start</button>
                </div>
            `).join('');
        }

        function saveQuiz() {
            const title = document.getElementById('quiz-title').value;
            const text = document.querySelector('.q-text').value;
            if(!title || !text) return alert("Please fill out the fields!");

            const newQuiz = {
                id: Date.now(),
                title: title,
                questions: [{
                    text: text,
                    a: document.querySelector('.opt-a').value,
                    b: document.querySelector('.opt-b').value,
                    correct: document.querySelector('.correct-opt').value.toUpperCase()
                }]
            };
            quizzes.push(newQuiz);
            alert("Quiz Published Successfully!");
            showView('home-view');
        }

        function startQuiz(id) {
            const quiz = quizzes.find(q => q.id === id);
            const container = document.getElementById('active-quiz-card');
            const q = quiz.questions[0]; // Simplified for demo to 1 question
            
            container.innerHTML = `
                <h3 style="margin-bottom: 1rem;">${quiz.title}</h3>
                <p style="font-size: 1.2rem; margin-bottom: 2rem;">${q.text}</p>
                <button class="option-btn" onclick="checkAnswer('A', '${q.correct}')">A: ${q.a}</button>
                <button class="option-btn" onclick="checkAnswer('B', '${q.correct}')">B: ${q.b}</button>
            `;
            showView('take-view');
        }

        function checkAnswer(selected, correct) {
            if(selected === correct) {
                alert("Correct! Score: 100%");
            } else {
                alert("Incorrect. The right answer was " + correct);
            }
            showView('home-view');
        }

        // Initial Load
        renderQuizList();
    </script>
